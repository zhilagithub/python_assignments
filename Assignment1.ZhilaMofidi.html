with open('emailSpam.txt', 'r') as file:

# Find the email subject using regular expressions
subject_match = re.search(r"Subject:\s*(.*)", content)
if subject_match:
    subject = subject_match.group(1)
    print("Email Subject:", subject)

    # Count the number of uppercase and lowercase letters in the subject
    num_uppercase = sum(1 for char in subject if char.isupper())
    num_lowercase = sum(1 for char in subject if char.islower())

    print("Number of uppercase letters:", num_uppercase)
    print("Number of lowercase letters:", num_lowercase)

spam_trigger_words = ['RENEWED', 'purchase', 'invoice', 'upgrade', 'cancel', 'auto-debited', 'subscription', 'customer service', 'billing', 'contact']

# Read the contents of the file
with open('emailSpam.txt', 'r') as file:
    content = file.read()

# Split the email body into individual words
words = content.split()

# Count the number of spam words found in the email body
spam_word_count = 0
for word in words:
    if word.lower() in spam_trigger_words:
        spam_word_count += 1

# Calculate the percentage of spam words
total_words = len(words)
percentage_spam_words = (spam_word_count / total_words) * 100

# Display the percentage of spam words
